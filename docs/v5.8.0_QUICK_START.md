# v5.8.0 Quick Start (5 Minutes)

## 1. Add Scripts (30 seconds)

Add to your `index.html`:
```html
<script src='assets/js/vr-wasm-simd-accelerator.js'></script>
<script src='assets/js/vr-advanced-gesture-dataset.js'></script>
<script src='assets/js/vr-gesture-customization.js'></script>
<script src='assets/js/vr-performance-dashboard.js'></script>
```

## 2. Initialize Core System (1 minute)

```javascript
// 1. Enable SIMD Acceleration (3-8x faster)
const accelerator = new VRWasmSimdAccelerator();
await accelerator.initialize();
await accelerator.optimizeGestureRecognition(gestureEngine);
console.log('SIMD enabled:', accelerator.simdSupported);

// 2. Load Extended Gestures (35 types)
const gestures = new VRAdvancedGestureDataset();
console.log('Gestures loaded:', gestures.gestures.size);

// 3. Setup Customization
const custom = new VRGestureCustomization(gestureEngine);
await custom.initialize();

// 4. Add Performance Dashboard
const dashboard = new VRPerformanceDashboard(profiler);
await dashboard.initialize();
```

## 3. Use Advanced Gestures (1 minute)

```javascript
// Get a gesture
const indexPoint = gestures.getGesture('index-point');
console.log('Confidence:', indexPoint.confidence); // 0.95

// Get a gesture family
const controls = gestures.getFamily('control');
console.log('Control gestures:', controls.length); // 10

// Get high-confidence gestures
const reliable = gestures.getGesturesByConfidence(0.9, 1.0);
console.log('Reliable gestures:', reliable.length);
```

## 4. Record a Macro (1.5 minutes)

```javascript
// Start recording
custom.startMacroRecording('my-combo');

// User performs gestures (the system listens)
// After 5 seconds of inactivity, auto-stops, or:
custom.stopMacroRecording();

// Check what was recorded
const macros = custom.listMacros();
console.log('Recorded macros:', macros);

// Execute the macro
custom.executeMacro('my-combo');
```

## 5. Create a Profile (1 minute)

```javascript
// Create a profile
custom.createProfile('gaming');
custom.switchProfile('gaming');

// Bind gestures to actions
custom.bindGestureToAction('index-point', 'jump');
custom.bindGestureToAction('peace-sign', 'attack');
custom.bindGestureToAction('my-combo', 'ultimate');

// When user makes a gesture, the action is triggered
const action = custom.getGestureBinding('index-point');
console.log('Action:', action); // 'jump'
```

## Complete Example

```html
<!DOCTYPE html>
<html>
<head>
    <title>VR Qui Browser v5.8.0</title>
    <script src='assets/js/vr-wasm-simd-accelerator.js'></script>
    <script src='assets/js/vr-advanced-gesture-dataset.js'></script>
    <script src='assets/js/vr-gesture-customization.js'></script>
    <script src='assets/js/vr-performance-dashboard.js'></script>
</head>
<body>
    <h1>v5.8.0 Example</h1>
    <div id='stats'></div>
    <script>
        async function init() {
            // Setup
            const acc = new VRWasmSimdAccelerator();
            await acc.initialize();
            
            const gest = new VRAdvancedGestureDataset();
            const custom = new VRGestureCustomization(window.gestureEngine);
            await custom.initialize();
            
            // Create profile
            custom.createProfile('demo');
            custom.switchProfile('demo');
            custom.bindGestureToAction('index-point', 'action1');
            
            // Record macro
            custom.startMacroRecording('test');
            setTimeout(() => {
                custom.stopMacroRecording();
                document.getElementById('stats').innerText = 
                    'Macros: ' + custom.listMacros().length;
            }, 2000);
        }
        
        init().catch(console.error);
    </script>
</body>
</html>
```
